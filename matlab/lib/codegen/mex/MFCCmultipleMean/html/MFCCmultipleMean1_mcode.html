1function [ mfcc ] = MFCCmultipleMean( mfccArIn )
 2%MFCCmultipleMean Computes the avg between all MFCC in cell array mfccAr
 3%   Takes the two closest elements, do the mean, iterate
 4mfccAr = mfccArIn;
 5n = size(mfccAr,1);
 6while (n>2)
 7    dtw = zeros(n);
 8    path = cell(n);
 9    path2 = coder.nullcopy(path);
10    %tic;
11    for i=1:n
12        for j=i:n
13            [dtw(i,j), path2{i,j}] = MFCCmatch(mfccAr{i},mfccAr{j});
14            dtw(j,i) = inf;
15            %path{j,i} = fliplr(path{i,j});
16            %disp(strcat('class-',int2str(i),' : class-',int2str(j),' done'));
17            %toc;
18        end
19    end
20    %toc;
21    %clearvars i j;
22    minInd = zeros(n,1);
23    minDtw = zeros(n,1);
24    tmp = zeros(n);
25    tmp((n+1)*(0:n-1)+1) = inf;
26    dtw = dtw + tmp;
27    for i=1:n
28        [minDtw(i), minInd(i)] = min(dtw(i,:));
29    end
30    mins = horzcat(minInd,minDtw);
31    [mins_sorted, indexes] = sortrows(mins,2);
32    mins_sorted = horzcat(indexes,mins_sorted);
33    mfccMeanClosest = MFCCmean(mfccAr{mins_sorted(1,1)},...
34        mfccAr{mins_sorted(1,2)},path2{mins_sorted(1,1),mins_sorted(1,2)});
35    mfccAr{mins_sorted(1,1)} = mfccMeanClosest;
36    mfccAr(mins_sorted(1,2)) = [];
37    n = size(mfccAr,1);
38end
39%clearvars tmp i j;
40[~, path] = MFCCmatch(mfccAr{1},mfccAr{2});
41mfcc = MFCCmean(mfccAr{1}, mfccAr{2},path);
42%{
43maxLength = 0;
44for i=1:size(mfccAr,1)
45    if maxLength<size(mfccAr{i},1)
46        maxLength = size(mfccAr{i},1);
47    end
48end
49for i=1:size(mfccAr,1)
50    mfccAr{i} = MFCCreduce(mfccAr{i},maxLength);
51end
52mfcc3 = zeros(size(mfccAr{i},1),size(mfccAr{i},2),length(mfccAr));
53for i=1:size(mfcc3,3)
54        mfcc3(:,:,i) = mfccAr{i};
55end
56mfcc = mean(mfcc3,3);
57end
58%}
